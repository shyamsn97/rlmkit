export const SPEED = 240;

const PERCEPTION_RADIUS = 60;
const SEPARATION_RADIUS = 18;
const ALIGNMENT_WEIGHT = 1.0;
const COHESION_WEIGHT = 0.6;
const SEPARATION_WEIGHT = 1.8;
const STEER = 140;

function normalize(x, y) {
  const m = Math.hypot(x, y) || 1;
  return [x / m, y / m];
}

export class Boid {
  constructor(x, y, angle, hue) {
    this.x = x;
    this.y = y;
    this.vx = Math.cos(angle) * SPEED;
    this.vy = Math.sin(angle) * SPEED;
    this.hue = hue;
    this.size = 3.5;
  }

  update(dt, boids, w, h) {
    let alignX = 0, alignY = 0;
    let centerX = 0, centerY = 0;
    let sepX = 0, sepY = 0;
    let count = 0;

    for (const other of boids) {
      if (other === this) continue;
      let dx = other.x - this.x;
      let dy = other.y - this.y;

      if (dx > w * 0.5) dx -= w; else if (dx < -w * 0.5) dx += w;
      if (dy > h * 0.5) dy -= h; else if (dy < -h * 0.5) dy += h;

      const d2 = dx * dx + dy * dy;

      if (d2 < PERCEPTION_RADIUS * PERCEPTION_RADIUS) {
        count++;
        alignX += other.vx;
        alignY += other.vy;
        centerX += this.x + dx;
        centerY += this.y + dy;

        if (d2 < SEPARATION_RADIUS * SEPARATION_RADIUS && d2 > 1e-6) {
          const inv = 1 / Math.sqrt(d2);
          sepX -= dx * inv;
          sepY -= dy * inv;
        }
      }
    }

    let steerX = 0, steerY = 0;
    if (count > 0) {
      let [ax, ay] = normalize(alignX, alignY);

      const cx = centerX / count - this.x;
      const cy = centerY / count - this.y;
      let [coX, coY] = normalize(cx, cy);

      let [sx, sy] = normalize(sepX, sepY);

      steerX = ALIGNMENT_WEIGHT * ax + COHESION_WEIGHT * coX + SEPARATION_WEIGHT * sx;
      steerY = ALIGNMENT_WEIGHT * ay + COHESION_WEIGHT * coY + SEPARATION_WEIGHT * sy;
    }

    this.vx += steerX * STEER * dt;
    this.vy += steerY * STEER * dt;

    const m = Math.hypot(this.vx, this.vy) || 1;
    this.vx = (this.vx / m) * SPEED;
    this.vy = (this.vy / m) * SPEED;

    this.x += this.vx * dt;
    this.y += this.vy * dt;

    if (this.x < 0) this.x += w; else if (this.x >= w) this.x -= w;
    if (this.y < 0) this.y += h; else if (this.y >= h) this.y -= h;
  }

  draw(ctx) {
    const angle = Math.atan2(this.vy, this.vx);
    const s = this.size;
    ctx.save();
    ctx.translate(this.x, this.y);
    ctx.rotate(angle);
    ctx.beginPath();
    ctx.moveTo(s * 2.2, 0);
    ctx.lineTo(-s, s);
    ctx.lineTo(-s, -s);
    ctx.closePath();
    ctx.fillStyle = `hsl(${this.hue}, 90%, 60%)`;
    ctx.fill();
    ctx.restore();
  }
}
